def tri(n):
    """Everyone knows Fibonacci sequence, it was studied deeply by mathematicians in 
    the last couple centuries. However, what people don't know is Tribonacci sequence.
    Tribonacci sequence is defined by the recurrence:
    tri(1) = 3
    tri(n) = 1 + n / 2, if n is even.
    tri(n) =  tri(n - 1) + tri(n - 2) + tri(n + 1), if n is odd.
    For example:
    tri(2) = 1 + (2 / 2) = 2
    tri(4) = 3
    tri(3) = tri(2) + tri(1) + tri(4)
           = 2 + 3 + 3 = 8 
    You are given a non-negative integer number n, you have to a return a list of the 
    first n + 1 numbers of the Tribonacci sequence.
    Examples:
    tri(3) = [1, 3, 2, 8]
    """
    if n < 0:
        return []
    
    # Initialize memoization dictionary with base cases
    memo = {0: 1, 1: 3}
    
    def calculate_tri(k):
        if k in memo:
            return memo[k]
        if k % 2 == 0:
            memo[k] = 1 + k // 2
        else:
            # For odd k, we need tri(k+1) which might not be calculated yet
            # So we need to calculate it recursively
            memo[k] = calculate_tri(k - 1) + calculate_tri(k - 2) + calculate_tri(k + 1)
        return memo[k]
    
    # Calculate all values up to n
    for i in range(2, n + 1):
        calculate_tri(i)
    
    # Return the sequence from 0 to n
    return [memo[i] for i in range(n + 1)]
