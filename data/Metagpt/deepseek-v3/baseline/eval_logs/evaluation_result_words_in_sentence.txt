def is_prime(n):
    """Helper function to check if a number is prime"""
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True

def words_in_sentence(sentence):
    """
    Filters words with prime lengths from a sentence and returns them in original order.
    
    Args:
        sentence: A string containing words separated by spaces
        
    Returns:
        A string containing words with prime lengths in original order
        
    Examples:
        >>> words_in_sentence("This is a test")
        'is'
        >>> words_in_sentence("lets go for swimming")
        'go for'
    """
    words = sentence.split()
    prime_words = [word for word in words if is_prime(len(word))]
    return ' '.join(prime_words)
