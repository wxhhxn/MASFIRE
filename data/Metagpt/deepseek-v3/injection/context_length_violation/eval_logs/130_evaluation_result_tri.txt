def tri(n):
    """
    Generate the first n+1 numbers of the Tribonacci sequence with special rules:
    - tri(1) = 3
    - If n is even: tri(n) = 1 + n / 2
    - If n is odd: tri(n) = tri(n-1) + tri(n-2) + tri(n+1)
    Returns a list of the sequence.
    """
    if n < 0:
        return []
    
    sequence = [1]  # tri(0) is 1 by the examples
    if n == 0:
        return sequence
        
    sequence.append(3)  # tri(1) = 3
    if n == 1:
        return sequence
    
    for i in range(2, n + 1):
        if i % 2 == 0:
            # Even case: 1 + n/2
            sequence.append(1 + i // 2)
        else:
            # Odd case: tri(n-1) + tri(n-2) + tri(n+1)
            # For n+1 we need to calculate it if not exists yet
            if i + 1 > len(sequence) - 1:
                # Calculate tri(n+1) temporarily
                next_val = 1 + (i + 1) // 2 if (i + 1) % 2 == 0 else 0
                sequence.append(sequence[i-1] + sequence[i-2] + next_val)
            else:
                sequence.append(sequence[i-1] + sequence[i-2] + sequence[i+1])
    
    return sequence[:n+1]
