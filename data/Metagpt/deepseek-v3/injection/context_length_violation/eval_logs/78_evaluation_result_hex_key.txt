def hex_key(num):
    """Count the number of prime hexadecimal digits in the given string.
    Prime hexadecimal digits are 2, 3, 5, 7, B, D.
    """
    primes = {'2', '3', '5', '7', 'B', 'D'}
    return sum(1 for digit in num if digit in primes)
