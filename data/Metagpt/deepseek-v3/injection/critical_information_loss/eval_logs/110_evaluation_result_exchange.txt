def exchange(lst1, lst2):
    """Determine if lst1 can be made all even by swapping elements with lst2.
    
    Args:
        lst1: List of integers to be made all even
        lst2: List of integers available for swapping
        
    Returns:
        "YES" if possible to make lst1 all even through swaps, "NO" otherwise
    """
    # Count the number of odd numbers in lst1
    odd_in_lst1 = sum(1 for num in lst1 if num % 2 != 0)
    
    # Count the number of even numbers in lst2 that could replace odd numbers in lst1
    even_in_lst2 = sum(1 for num in lst2 if num % 2 == 0)
    
    # We need enough even numbers in lst2 to replace all odd numbers in lst1
    if odd_in_lst1 <= even_in_lst2:
        return "YES"
    else:
        return "NO"
