def starts_one_ends(n):
    """
    Given a positive integer n, return the count of the numbers of n-digit
    positive integers that start or end with 1.
    """
    if n == 1:
        return 1
    
    # Count of numbers starting with 1: 1 * 10^(n-1)
    start_with_1 = 10 ** (n - 1)
    
    # Count of numbers ending with 1: 9 * 10^(n-2) * 1 (last digit fixed as 1)
    end_with_1 = 9 * 10 ** (n - 2)
    
    # Count of numbers both starting and ending with 1 (to avoid double counting)
    both = 10 ** (n - 2) if n >= 2 else 0
    
    return start_with_1 + end_with_1 - both
