from typing import List

def solution(lst: List[int]) -> int:
    """Given a non-empty list of integers, return the sum of all of the odd elements that are in even positions.
    
    Uses 1-based indexing where even positions are 1, 3, 5, etc.
    If list length is odd, appends a 0 to make it even length before processing.
    
    Examples:
    >>> solution([5, 8, 7, 1])
    12
    >>> solution([3, 3, 3, 3, 3])
    9
    >>> solution([30, 13, 24, 321])
    0
    """
    # Make sure list is even length
    if len(lst) % 2 != 0:
        lst = lst + [0]
    
    total = 0
    # Iterate over even positions (1-based: 1, 3, 5...)
    for i in range(0, len(lst), 2):
        if lst[i] % 2 != 0:  # Check if odd
            total += lst[i]
    return total
