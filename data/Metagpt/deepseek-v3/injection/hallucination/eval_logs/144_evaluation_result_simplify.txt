def simplify(x, n):
    """Check if product of two fractions is a whole number.
    
    Args:
        x: First fraction as string (format 'a/b')
        n: Second fraction as string (format 'c/d')
        
    Returns:
        bool: True if x * n is whole number, False otherwise
    """
    # Parse first fraction
    a, b = map(int, x.split('/'))
    # Parse second fraction
    c, d = map(int, n.split('/'))
    
    # Calculate numerator and denominator of product
    numerator = a * c
    denominator = b * d
    
    # Check if the product is a whole number
    return (numerator % denominator) == 0
