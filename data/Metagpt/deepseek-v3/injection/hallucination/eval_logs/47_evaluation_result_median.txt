def median(l: list) -> float:
    """Return median of elements in the list l.
    For even-length lists, returns the lower (left) middle element.
    Assumes input is a non-empty list of integers between 0 and 100.
    """
    sorted_l = sorted(l)
    n = len(sorted_l)
    if n % 2 == 1:
        return sorted_l[n // 2]
    else:
        return (sorted_l[n // 2 - 1] + sorted_l[n // 2]) / 2
