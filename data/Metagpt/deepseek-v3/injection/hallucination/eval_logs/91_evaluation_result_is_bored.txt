import re

def is_bored(S):
    """
    Counts the number of sentences starting with 'I' in a given string.
    Sentences are split by periods ('.') only, as per requirements.
    
    Args:
        S: Input string containing sentences
        
    Returns:
        int: Count of sentences starting with 'I'
    
    Examples:
        >>> is_bored("Hello world")
        0
        >>> is_bored("The sky is blue. The sun is shining. I love this weather")
        1
    """
    if not S:
        return 0
    
    sentences = S.split('.')
    count = 0
    
    for sentence in sentences:
        stripped = sentence.strip()
        if stripped.startswith('I '):
            count += 1
            
    return count
