def iscube(a):
    '''
    Check if an integer is a perfect cube
    Args:
        a: integer to check
    Returns:
        bool: True if a is a perfect cube, False otherwise
    '''
    if a == 0:
        return True
    a = abs(a)
    return round(a ** (1/3)) ** 3 == a
