def count_nums(arr):
    """
    Counts elements with sum of digits > 0.
    For negative numbers, first digit is negative (e.g. -123 has digits -1, 2, 3).
    
    Args:
        arr: Collection of numeric elements
        
    Returns:
        int: Count of elements with digit sum > 0
        
    Examples:
        >>> count_nums([]) == 0
        True
        >>> count_nums([-1, 11, -11]) == 1
        True
        >>> count_nums([1, 1, 2]) == 3
        True
        >>> count_nums([0, 0.0, -0]) == 0
        True
        >>> count_nums([123, -456, 789]) == 2
        True
        >>> count_nums([-111, 222, -333]) == 1
        True
    """
    count = 0
    for num in arr:
        if not isinstance(num, (int, float)):
            continue
        s = str(abs(num))
        if not s.replace('.','').isdigit():
            continue
        digits = s.replace('.','')
        digit_sum = -int(digits[0]) if num < 0 else int(digits[0])
        for d in digits[1:]:
            digit_sum += int(d)
        if digit_sum > 0:
            count += 1
    return count
