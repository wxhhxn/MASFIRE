def sorted_list_sum(lst):
    """Filters out tokens with odd lengths and returns a sorted list.
    
    Args:
        lst: A collection of textual tokens (strings)
        
    Returns:
        A list of tokens with even lengths, sorted by length (ascending) and
        alphabetically for tokens of the same length.
    """
    # Filter out tokens with odd lengths
    even_length_tokens = [token for token in lst if len(token) % 2 == 0]
    
    # Sort first by length (ascending), then alphabetically
    sorted_tokens = sorted(even_length_tokens, key=lambda x: (len(x), x))
    
    return sorted_tokens
