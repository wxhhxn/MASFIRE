def search(lst):
    '''
    You are given a non-empty collection of positive numeric elements. Return the greatest element that is greater than 
    zero, and has a frequency greater than or equal to the value of the element itself. 
    The frequency of an element is the number of times it appears in the collection.
    If no such a value exist, return -1.
    Examples:
        search([4, 1, 2, 2, 3, 1]) == 2
        search([1, 2, 2, 3, 3, 3, 4, 4, 4]) == 3
        search([5, 5, 4, 4, 4]) == -1
    '''
    if not lst:
        return -1
    
    freq = defaultdict(int)
    for num in lst:
        if num > 0:
            freq[num] += 1
    
    result = -1
    for num, count in freq.items():
        if num <= count and num > result:
            result = num
    
    return result
