def hex_key(num: str) -> int:
    """Count the number of prime digits in a hexadecimal string.
    
    Args:
        num: Hexadecimal string to analyze
        
    Returns:
        Count of prime digits in the string
    """
    count = 0
    for digit in num:
        if digit in PRIME_DIGITS:
            count += 1
    return count
