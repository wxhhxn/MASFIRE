def get_max_triples(n):
    """
    Given a positive integer n, creates an array a of length n where a[i] = i*i - i + 1.
    Returns the number of triples (a[i], a[j], a[k]) where i < j < k and a[i] + a[j] + a[k]
    is divisible by 3.
    
    Args:
        n: positive integer length of array
    
    Returns:
        int: count of valid triples
    
    Example:
        >>> get_max_triples(5)
        1
    """
    if n < 3:
        return 0
    
    a = [i*i - i + 1 for i in range(1, n+1)]
    
    # Count remainders when divided by 3
    remainder_counts = [0, 0, 0]  # for remainders 0, 1, 2
    
    for num in a:
        remainder = num % 3
        remainder_counts[remainder] += 1
    
    # Possible valid combinations:
    # 1) All 3 with remainder 0
    # 2) One from each remainder (0,1,2)
    # 3) All 3 with remainder 1
    # 4) All 3 with remainder 2
    count = 0
    
    # Combination 1: choose 3 from remainder 0
    count += remainder_counts[0] * (remainder_counts[0] - 1) * (remainder_counts[0] - 2) // 6 if remainder_counts[0] >= 3 else 0
    
    # Combination 2: choose one from each remainder
    count += remainder_counts[0] * remainder_counts[1] * remainder_counts[2]
    
    # Combination 3: choose 3 from remainder 1
    count += remainder_counts[1] * (remainder_counts[1] - 1) * (remainder_counts[1] - 2) // 6 if remainder_counts[1] >= 3 else 0
    
    # Combination 4: choose 3 from remainder 2
    count += remainder_counts[2] * (remainder_counts[2] - 1) * (remainder_counts[2] - 2) // 6 if remainder_counts[2] >= 3 else 0
    
    return count
