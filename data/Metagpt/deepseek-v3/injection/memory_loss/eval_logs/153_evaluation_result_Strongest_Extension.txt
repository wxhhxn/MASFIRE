def Strongest_Extension(class_name, extensions):
    """Find the strongest extension from a list based on uppercase/lowercase letter counts.
    
    Args:
        class_name: Name of the class (string)
        extensions: List of extension names (list of strings)
        
    Returns:
        String in format 'ClassName.StrongestExtensionName'
    """
    if not extensions:
        return f"{class_name}."
        
    strongest_extension = None
    max_strength = -float('inf')
    
    for ext in extensions:
        cap = sum(1 for c in ext if c.isupper())
        sm = sum(1 for c in ext if c.islower())
        strength = cap - sm
        
        if strength > max_strength:
            max_strength = strength
            strongest_extension = ext
    
    return f"{class_name}.{strongest_extension}"
