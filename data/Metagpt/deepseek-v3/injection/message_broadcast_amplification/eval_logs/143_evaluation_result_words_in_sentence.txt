def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def words_in_sentence(sentence):
    words = sentence.split()
    result_words = []
    for word in words:
        if is_prime(len(word)):
            result_words.append(word)
    return ' '.join(result_words)
