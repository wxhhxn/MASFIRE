def total_match(lst1, lst2):
    """
    Compare total characters in two string lists and return the list with fewer total characters.
    If equal, return the first list.
    
    Args:
        lst1 (list): First list of strings
        lst2 (list): Second list of strings
        
    Returns:
        list: The list with fewer total characters, or lst1 if equal
    """
    def count_chars(string_list):
        """Helper function to count total characters in a list of strings"""
        return sum(len(s) for s in string_list)
    
    total1 = count_chars(lst1)
    total2 = count_chars(lst2)
    
    if total1 <= total2:
        return lst1
    else:
        return lst2
