import math

def f(n):
    result = []
    for i in range(1, n + 1):
        if i % 2 == 0:
            # Calculate factorial for even i
            factorial = 1
            for j in range(1, i + 1):
                factorial *= j
            result.append(factorial)
        else:
            # Calculate sum for odd i
            sum_val = 0
            for j in range(1, i + 1):
                sum_val += j
            result.append(sum_val)
    return result
