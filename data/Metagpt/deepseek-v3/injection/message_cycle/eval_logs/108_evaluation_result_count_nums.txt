def count_nums(arr):
    """
    Write a function count_nums which takes an array of integers and returns
    the number of elements which has a sum of digits > 0.
    If a number is negative, then its first signed digit will be negative:
    e.g. -123 has signed digits -1, 2, and 3.
    >>> count_nums([]) == 0
    >>> count_nums([-1, 11, -11]) == 1
    >>> count_nums([1, 1, 2]) == 3
    """
    def digit_sum(n):
        if n == 0:
            return 0
        if n < 0:
            # Convert to positive for digit processing
            n = abs(n)
            # Get the first digit and make it negative
            first_digit = int(str(n)[0])
            remaining_digits = str(n)[1:]
            total = -first_digit
            # Add remaining digits
            for digit in remaining_digits:
                total += int(digit)
            return total
        else:
            # Positive number case
            total = 0
            for digit in str(n):
                total += int(digit)
            return total

    count = 0
    for num in arr:
        if digit_sum(num) > 0:
            count += 1
    return count
