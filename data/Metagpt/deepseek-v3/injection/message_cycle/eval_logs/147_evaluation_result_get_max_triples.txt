def get_max_triples(n):
    """
    You are given a positive integer n. You have to create an integer array a of length n.
        For each i (1 ≤ i ≤ n), the value of a[i] = i * i - i + 1.
        Return the number of triples (a[i], a[j], a[k]) of a where i < j < k, 
    and a[i] + a[j] + a[k] is a multiple of 3.

    Example :
        Input: n = 5
        Output: 1
        Explanation: 
        a = [1, 3, 7, 13, 21]
        The only valid triple is (1, 7, 13).
    """
    # Count remainders modulo 3
    count = [0, 0, 0]  # count[0], count[1], count[2] for remainders 0, 1, 2
    
    for i in range(1, n + 1):
        # Calculate a[i] = i² - i + 1
        value = i * i - i + 1
        remainder = value % 3
        count[remainder] += 1
    
    # The sum of three numbers is divisible by 3 if:
    # 1. All three have remainder 0
    # 2. All three have remainder 1 (1+1+1=3≡0 mod 3)
    # 3. All three have remainder 2 (2+2+2=6≡0 mod 3)
    # 4. One of each remainder (0+1+2=3≡0 mod 3)
    
    # Calculate combinations
    total = 0
    
    # Case 1: All three with remainder 0
    if count[0] >= 3:
        total += count[0] * (count[0] - 1) * (count[0] - 2) // 6
    
    # Case 2: All three with remainder 1
    if count[1] >= 3:
        total += count[1] * (count[1] - 1) * (count[1] - 2) // 6
    
    # Case 3: All three with remainder 2
    if count[2] >= 3:
        total += count[2] * (count[2] - 1) * (count[2] - 2) // 6
    
    # Case 4: One of each remainder
    total += count[0] * count[1] * count[2]
    
    return total
