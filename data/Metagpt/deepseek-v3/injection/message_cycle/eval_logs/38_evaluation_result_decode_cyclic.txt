def decode_cyclic(s: str) -> str:
    """
    Reverses the encoding process of encode_cyclic function.
    Takes a string encoded with encode_cyclic and returns the original string.
    """
    # Split the string into groups of 3 characters
    groups = [s[i:i+3] for i in range(0, len(s), 3)]
    
    # Reverse the cyclic shift for each group
    # For groups of length 3: move last character to the front
    # For groups with fewer than 3 characters: leave unchanged
    decoded_groups = []
    for group in groups:
        if len(group) == 3:
            # Move the last character to the front to reverse the encoding
            decoded_group = group[-1] + group[:-1]
            decoded_groups.append(decoded_group)
        else:
            decoded_groups.append(group)
    
    return "".join(decoded_groups)
