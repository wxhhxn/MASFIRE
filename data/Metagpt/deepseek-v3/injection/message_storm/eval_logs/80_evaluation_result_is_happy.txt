def is_happy(s):
    """Check if a string is happy.
    
    A string is happy if its length is at least 3 and every 3 consecutive letters are distinct.
    
    Args:
        s (str): The input string to check
        
    Returns:
        bool: True if the string is happy, False otherwise
        
    Examples:
        >>> is_happy('a')
        False
        >>> is_happy('aa')
        False
        >>> is_happy('abcd')
        True
        >>> is_happy('aabb')
        False
        >>> is_happy('adb')
        True
        >>> is_happy('xyy')
        False
    """
    # Check if length is less than 3
    if len(s) < 3:
        return False
        
    # Check every 3 consecutive characters
    for i in range(len(s) - 2):
        # Get the current triplet
        triplet = s[i:i+3]
        # Check if all three characters are distinct
        if len(set(triplet)) != 3:
            return False
            
    return True
