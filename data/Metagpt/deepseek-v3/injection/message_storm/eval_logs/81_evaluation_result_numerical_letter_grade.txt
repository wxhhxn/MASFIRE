def numerical_letter_grade(grades):
    """Convert a list of GPAs to letter grades according to the specified grading scale.
    
    Args:
        grades: List of numerical GPA values
        
    Returns:
        List of letter grades corresponding to the input GPAs
        
    Grading Scale:
        4.0 = A+
        >3.7 = A
        >3.3 = A-
        >3.0 = B+
        >2.7 = B
        >2.3 = B-
        >2.0 = C+
        >1.7 = C
        >1.3 = C-
        >1.0 = D+
        >0.7 = D
        >0.0 = D-
        0.0 = E
    """
    letter_grades = []
    
    for gpa in grades:
        if gpa == 4.0:
            letter_grades.append('A+')
        elif gpa > 3.7:
            letter_grades.append('A')
        elif gpa > 3.3:
            letter_grades.append('A-')
        elif gpa > 3.0:
            letter_grades.append('B+')
        elif gpa > 2.7:
            letter_grades.append('B')
        elif gpa > 2.3:
            letter_grades.append('B-')
        elif gpa > 2.0:
            letter_grades.append('C+')
        elif gpa > 1.7:
            letter_grades.append('C')
        elif gpa > 1.3:
            letter_grades.append('C-')
        elif gpa > 1.0:
            letter_grades.append('D+')
        elif gpa > 0.7:
            letter_grades.append('D')
        elif gpa > 0.0:
            letter_grades.append('D-')
        else:
            letter_grades.append('E')
    
    return letter_grades
