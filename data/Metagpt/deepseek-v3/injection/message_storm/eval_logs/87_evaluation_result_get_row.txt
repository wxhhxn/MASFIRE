def get_row(lst, x):
    """
    You are given a 2 dimensional data, as a nested lists,
    which is similar to matrix, however, unlike matrices,
    each row may contain a different number of columns.
    Given lst, and integer x, find integers x in the list,
    and return list of tuples, [(x1, y1), (x2, y2) ...] such that
    each tuple is a coordinate - (row, columns), starting with 0.
    Sort coordinates initially by rows in ascending order.
    Also, sort coordinates of the row by columns in descending order.
    
    Examples:
    get_row([
      [1,2,3,4,5,6],
      [1,2,3,4,1,6],
      [1,2,3,4,5,1]
    ], 1) == [(0, 0), (1, 4), (1, 0), (2, 5), (2, 0)]
    get_row([], 1) == []
    get_row([[], [1], [1, 2, 3]], 3) == [(2, 2)]
    """
    result = []
    for row_idx, row in enumerate(lst):
        # Collect all column indices where value equals x in this row
        columns = []
        for col_idx, value in enumerate(row):
            if value == x:
                columns.append(col_idx)
        
        # Sort column indices in descending order for this row
        columns.sort(reverse=True)
        
        # Add coordinates to result
        for col_idx in columns:
            result.append((row_idx, col_idx))
    
    # The result is already sorted by row in ascending order due to enumerate
    return result
