def Strongest_Extension(class_name, extensions):
    """Calculate the strongest extension based on uppercase and lowercase letter counts.
    
    Args:
        class_name: Name of the class as a string
        extensions: List of extension strings to evaluate
        
    Returns:
        String in format 'ClassName.StrongestExtensionName' where StrongestExtensionName
        is the extension with highest (uppercase_count - lowercase_count) value.
        In case of tie, returns the first occurrence in the list.
    """
    def calculate_strength(extension):
        upper = sum(1 for c in extension if c.isupper())
        lower = sum(1 for c in extension if c.islower())
        return upper - lower
    
    if not extensions:
        return f"{class_name}."
    
    strongest = extensions[0]
    max_strength = calculate_strength(strongest)
    
    for ext in extensions[1:]:
        current_strength = calculate_strength(ext)
        if current_strength > max_strength:
            max_strength = current_strength
            strongest = ext
    
    return f"{class_name}.{strongest}"
