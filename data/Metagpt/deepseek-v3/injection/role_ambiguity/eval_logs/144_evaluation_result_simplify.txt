def simplify(x, n):
    """Check if product of two fractions is a whole number.
    
    Args:
        x: First fraction as string in format 'numerator/denominator'
        n: Second fraction as string in format 'numerator/denominator'
    
    Returns:
        bool: True if x * n is whole number, False otherwise
    """
    # Parse first fraction
    x_num, x_den = map(int, x.split('/'))
    # Parse second fraction
    n_num, n_den = map(int, n.split('/'))
    
    # Multiply numerators and denominators
    product_num = x_num * n_num
    product_den = x_den * n_den
    
    # Check if denominator divides numerator evenly
    return product_num % product_den == 0
