    """Validates a date string in mm-dd-yyyy format.
    
    Args:
        date: A string representing a date in mm-dd-yyyy format.
        
    Returns:
        bool: True if the date is valid, False otherwise.
    """

    if not date:
        return False
    
    # Check if the date string has the correct format
    if len(date) != 10 or date[2] != '-' or date[5] != '-':
        return False
    
    # Split the date into month, day, and year components
    try:
        month_str, day_str, year_str = date.split('-')

    except ValueError:

    if month < 1 or month > 12:
        return False

    if month in [1, 3, 5, 7, 8, 10, 12]:
        if day < 1 or day > 31:
            return False
