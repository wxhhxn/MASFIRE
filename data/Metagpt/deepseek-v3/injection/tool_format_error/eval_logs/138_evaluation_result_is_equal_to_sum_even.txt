def is_equal_to_sum_even(n):
    """Evaluate whether the given number n can be written as the sum of exactly 4 positive even numbers
    Example
    is_equal_to_sum_even(4) == False
    is_equal_to_sum_even(6) == False
    is_equal_to_sum_even(8) == True
    """
    # The smallest sum of 4 positive even numbers is 2+2+2+2 = 8
    if n < 8:
        return False
    
    # For n to be the sum of 4 even numbers, n must be even
    # (since sum of even numbers is always even)
    if n % 2 != 0:
        return False
    
    # Any even number >= 8 can be expressed as the sum of 4 positive even numbers
    # We can represent n as: n = 2 + 2 + 2 + (n-6)
    # Since n >= 8, (n-6) >= 2, which is positive and even
    return True
