def cycpattern_check(a, b):
    """Check if any rotation of string b is a substring of string a.
    
    Args:
        a: The main string to search within
        b: The string whose rotations will be checked against a
        
    Returns:
        bool: True if any rotation of b is found in a, False otherwise
        
    Examples:
        >>> cycpattern_check("abcd","abd")
        False
        >>> cycpattern_check("hello","ell")
        True
        >>> cycpattern_check("whassup","psus")
        False
        >>> cycpattern_check("abab","baa")
        True
        >>> cycpattern_check("efef","eeff")
        False
        >>> cycpattern_check("himenss","simen")
        True
    """
    if not b:
        return True
    
    doubled_b = b + b
    for i in range(len(b)):
        rotated = doubled_b[i:i+len(b)]
        if rotated in a:
            return True
    return False
