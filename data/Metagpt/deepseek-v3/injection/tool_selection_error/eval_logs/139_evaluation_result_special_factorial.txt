def brazilian_factorial(n: int) -> int:
    """Compute the factorial of a non-negative integer n.

    Args:
        n: Non-negative integer

    Returns:
        n! as an integer
    """
    if not isinstance(n, int):
        raise TypeError("n must be an integer")
