'''Module to sort space-delimited number words from 'zero' to 'nine' in ascending order.'''

from typing import Dict

_NUMBER_ORDER: Dict[str, int] = {
    'zero': 0,
    'one': 1,
    'two': 2,
    'three': 3,
    'four': 4,
    'five': 5,
    'six': 6,
    'seven': 7,
    'eight': 8,
    'nine': 9,
}


def sort_numbers(s: str) -> str:
    """Return the input string with space-delimited number words sorted in ascending order.

    Examples:
        sort_numbers('one zero three') -> 'zero one three'

    Raises:
        ValueError: if the input contains tokens that are not valid number words
    """
    if not s:
        return ""

    tokens = s.split()
    try:
        sorted_tokens = sorted(tokens, key=lambda w: _NUMBER_ORDER[w])
    except KeyError as exc:
        raise ValueError(f"Invalid number word: {exc}") from exc

    return " ".join(sorted_tokens)
