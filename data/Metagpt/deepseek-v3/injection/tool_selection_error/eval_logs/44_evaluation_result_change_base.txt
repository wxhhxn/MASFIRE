# -*- coding: utf-8 -*-
'''Utility to convert integer to string representation in given base (<10).'''

def change_base(n: int, base: int) -> str:
    '''Convert integer n to a string in the given base (2 <= base <= 9).

    Negative numbers are prefixed with '-'.
    '''
    if not isinstance(base, int) or base < 2 or base >= 10:
        raise ValueError('base must be an integer between 2 and 9 inclusive')
    if not isinstance(n, int):
        raise TypeError('n must be an integer')
    if n == 0:
        return '0'
    neg = n < 0
    n = abs(n)
    digits = []
    while n:
        digits.append(str(n % base))
        n //= base
    if neg:
        digits.append('-')
    return ''.join(reversed(digits))
