"""Utility to remove vowels from a string.

This module provides a single function `remove_vowels` which returns a
copy of the input string with all ASCII vowels removed (both uppercase
and lowercase).
"""

def remove_vowels(s: str) -> str:
    """Return a copy of `s` with all vowels removed.

    Args:
        s: Input string.

    Returns:
        A new string with all occurrences of 'a', 'e', 'i', 'o', 'u'
        (case-insensitive) removed.

    Raises:
        TypeError: If the input is not a string.
    """
    if not isinstance(s, str):
        raise TypeError("Input must be a string")
    vowels = set("aeiouAEIOU")
    return "".join(ch for ch in s if ch not in vowels)
