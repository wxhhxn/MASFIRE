# -*- coding: utf-8 -*-
"""Utility to convert decimal to binary with 'db' prefix and suffix."""

def decimal_to_binary(n: int) -> str:
    """Convert a decimal integer to a binary representation wrapped with 'db' prefix and suffix.

    Examples:
        decimal_to_binary(5) -> 'db101db'
        decimal_to_binary(0) -> 'db0db'
    """
    if not isinstance(n, int):
        raise TypeError("Input must be an integer")
    if n == 0:
        return 'db0db'
    sign = '-' if n < 0 else ''
    n_abs = abs(n)
    bits = bin(n_abs)[2:]
    return f"db{sign}{bits}db"
