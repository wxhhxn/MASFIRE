def _is_prime(n: int) -> bool:
    """Return True if n is a prime number.

    Uses a simple deterministic check suitable for small integers.
    """
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0:
        return False
    i = 3
    while i * i <= n:
        if n % i == 0:
            return False
        i += 2
    return True
