def is_palindrome(string: str) -> bool:
    """Test if given string is a palindrome."""
    return string == string[::-1]


def make_palindrome(string: str) -> str:
    """
    Find the shortest palindrome that begins with a supplied string.

    Algorithm idea is simple:
    - Find the longest postfix (suffix) of the supplied string that is a palindrome.
    - Append to the end of the string the reverse of the string prefix that comes before the palindromic suffix.

    >>> make_palindrome('')
    ''
    >>> make_palindrome('cat')
    'catac'
    >>> make_palindrome('cata')
    'catac'
    """
    n = len(string)
    for i in range(n + 1):
        tail = string[i:]
        if is_palindrome(tail):
            return string + string[:i][::-1]
    # Should not be reached because the empty suffix is a palindrome
    return string
