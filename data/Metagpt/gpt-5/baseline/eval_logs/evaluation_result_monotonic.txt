def monotonic(l: list):
    """
    Return True if list elements are monotonically increasing or decreasing.

    Monotonic here means non-strict: each element <= next (increasing) or
    each element >= next (decreasing). Empty or single-element lists are monotonic.
    """
    n = len(l)
    if n <= 1:
        return True
    inc = True
    dec = True
    for i in range(n - 1):
        if l[i] > l[i + 1]:
            inc = False
        if l[i] < l[i + 1]:
            dec = False
        if not inc and not dec:
            return False
    return True
