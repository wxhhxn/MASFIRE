from typing import List, Union


def split_words(txt: str) -> Union[List[str], int]:
    """
    Split a string based on the rules:
    - If any whitespace exists, split on whitespace and return the list (txt.split()).
    - Else if any comma exists, split on commas and return the list (txt.split(',')).
    - Else return the count of lowercase letters with odd alphabet index (a=0 ... z=25).
    """
    # Check for any whitespace characters
    if any(ch.isspace() for ch in txt):
        return txt.split()
    # Check for comma
    if "," in txt:
        return txt.split(",")
    # Count lowercase letters whose index is odd
    count = 0
    for ch in txt:
        if "a" <= ch <= "z":
            if (ord(ch) - ord("a")) % 2 == 1:
                count += 1
    return count
