import re


def valid_date(date: str) -> bool:
    """
    Validate a date string in mm-dd-yyyy format per rules:
    - Non-empty string.
    - Month 01-12.
    - Day range depends on month; February allows 1-29 (no leap year check).
    - Year is any four digits.
    """
    if not isinstance(date, str) or not date:
        return False

    match = re.fullmatch(r"(\d{2})-(\d{2})-(\d{4})", date)
    if not match:
        return False

    month = int(match.group(1))
    day = int(match.group(2))
    # year = match.group(3)  # Year is any four digits; no further checks required.

    if month < 1 or month > 12:
        return False

    if month in {1, 3, 5, 7, 8, 10, 12}:
        return 1 <= day <= 31
    if month in {4, 6, 9, 11}:
        return 1 <= day <= 30
    # month == 2
    return 1 <= day <= 29
