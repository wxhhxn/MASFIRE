import math

def x_or_y(n: int, x, y):
    """
    Return x if n is a prime number; otherwise return y.
    """
    if n < 2:
        return y
    if n == 2:
        return x
    if n % 2 == 0:
        return y
    limit = math.isqrt(n)
    d = 3
    while d <= limit:
        if n % d == 0:
            return y
        d += 2
    return x
