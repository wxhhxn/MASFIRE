def any_int(x, y, z):
    """
    Returns True if at least one number equals the sum of the other two and at least one input is an integer; otherwise False.
    """
    if not any(isinstance(n, int) for n in (x, y, z)):
        return False
    return x == y + z or y == x + z or z == x + y

print(any_int(5, 2, 7))
