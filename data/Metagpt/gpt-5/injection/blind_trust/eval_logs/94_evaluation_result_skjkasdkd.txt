    For lst = [1,0,1,8,2,4597,2,1,3,40,1,2,1,2,4,2,5,1] the output should be 25
    For lst = [1,3,1,32,5107,34,83278,109,163,23,2323,32,30,1,9,3] the output should be 13

    def is_prime(n):
        if n == 1:
            return True
        if n <= 0:
            return False
        if n == 2:
            return True
        if n % 2 == 0:
            return False
        i = 3
        half = n // 2
        while i <= half:
            if n % i == 0:
                return False
            i += 2
        return True
