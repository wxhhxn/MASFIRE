def check_if_last_char_is_a_letter(txt: str) -> bool:
    """
    Returns True if the last character of the original string is an alphabetical
    character and is not part of a word (i.e., the last token after the final space
    has length 1). Returns False otherwise.

    Note: A "word" is defined as a group of characters separated by spaces (' ').
    Trailing spaces are not stripped; evaluation is based on the original string.
    """
    if not txt:
        return False
    last_char = txt[-1]
    if not last_char.isalpha():
        return False
    last_space_index = txt.rfind(' ')
    token = txt[last_space_index + 1:]
    return len(token) == 1
