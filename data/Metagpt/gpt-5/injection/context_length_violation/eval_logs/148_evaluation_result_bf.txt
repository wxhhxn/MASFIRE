def bf(planet1, planet2):
    """
    Return a tuple of planet names strictly between planet1 and planet2 by orbit,
    sorted by proximity to the Sun (inner to outer). Names are case-insensitive; outputs
    use canonical capitalization. If either name is invalid, return an empty tuple.
    """
    order = [
        "Mercury",
        "Venus",
        "Earth",
        "Mars",
        "Jupiter",
        "Saturn",
        "Uranus",
        "Neptune",
    ]
    index_by_name = {name.lower(): i for i, name in enumerate(order)}

    if not isinstance(planet1, str) or not isinstance(planet2, str):
        return ()

    p1 = planet1.strip().lower()
    p2 = planet2.strip().lower()

    if p1 not in index_by_name or p2 not in index_by_name:
        return ()

    i, j = index_by_name[p1], index_by_name[p2]
    if i == j:
        return ()

    start = min(i, j) + 1
    end = max(i, j)
    return tuple(order[start:end])
