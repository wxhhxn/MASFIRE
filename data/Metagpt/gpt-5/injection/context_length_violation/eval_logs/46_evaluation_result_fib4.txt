    """
    The Fib4 number sequence is a sequence similar to the Fibbonacci sequnece that's defined as follows:
    fib4(0) -> 0
    fib4(1) -> 0
    fib4(2) -> 2
    fib4(3) -> 0

    >>> fib4(5)
    4

    8
    >>> fib4(7)
    14
    """
    if n < 4:
        return [0, 0, 2, 0][n]
    a0, a1, a2, a3 = 0, 0, 2, 0
