def rounded_avg(n, m):
    """Compute the average of integers from n through m (inclusive), round to the nearest
    integer, and return the binary representation as a string (e.g., "0b101").

    If n > m, return -1.

    Notes:
    - Non-integer inputs raise TypeError.
    - Rounding follows Python's round semantics (ties to even), which matches examples
      such as 20..33 -> average 26.5 -> rounded 26 -> "0b11010".
    """
    if not isinstance(n, int) or not isinstance(m, int):
        raise TypeError("n and m must be integers")

    if n > m:
        return -1

    # The mean of a consecutive integer range [n, m] is (n + m) / 2.
    avg = (n + m) / 2
    return bin(int(round(avg)))
