def exchange(lst1, lst2):
    """Return "YES" if elements can be exchanged between lst1 and lst2 so that lst1 contains only even numbers; otherwise return "NO".

    The operation allows swapping any number of elements between the two lists. This is possible if and only if the total number of even
    elements in the combined multiset of lst1 and lst2 is at least the length of lst1.
    Assumes the lists are non-empty and elements support modulo operation by 2.
    """
    total_evens = sum(1 for x in (lst1 + lst2) if x % 2 == 0)
    return "YES" if total_evens >= len(lst1) else "NO"
