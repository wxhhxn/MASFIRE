from typing import List


def minSubArraySum(nums: List[int]) -> int:
    """
    Return the minimum sum of any non-empty contiguous sub-array of nums.

    Args:
        nums: A list of integers.

    Returns:
        The minimum sum among all non-empty contiguous sub-arrays.

    Raises:
        ValueError: If nums is empty.
    """
    if not nums:
        raise ValueError("nums must be a non-empty list")

    current_min = global_min = nums[0]
    for x in nums[1:]:
        current_min = min(x, current_min + x)
        if current_min < global_min:
            global_min = current_min
    return global_min
