import math


def max_fill(grid, capacity):
    """
    Compute the total number of bucket lowerings required to empty all wells.

    Each row in grid represents a single well. Each 1 in a row is one unit of water.
    With bucket capacity 'capacity', the number of lowerings for a row equals
    ceil(number_of_ones_in_row / capacity). The answer is the sum across rows.

    Parameters:
    - grid: List[List[int]]
    - capacity: int

    Returns:
    - int: total number of bucket lowerings.
    """
    if capacity <= 0:
        raise ValueError("capacity must be a positive integer")

    total = 0
    for row in grid:
        units = sum(row)
        if units:
            total += math.ceil(units / capacity)
    return total
