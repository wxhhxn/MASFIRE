def get_closest_vowel(word: str) -> str:
    """Return the closest vowel from the right that stands between two consonants.

    Vowels at the beginning or end do not count. Returns an empty string if none.
    """
    vowels = set("AEIOUaeiou")
    # Scan from right to left, skipping the first and last character
    for i in range(len(word) - 2, 0, -1):
        c = word[i]
        if c in vowels and word[i - 1] not in vowels and word[i + 1] not in vowels:
            return c
    return ""
