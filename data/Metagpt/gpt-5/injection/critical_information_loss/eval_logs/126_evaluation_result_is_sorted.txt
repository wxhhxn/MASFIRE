from collections import Counter


def is_sorted(lst):
    """
    Return True if the list of non-negative integers is in non-decreasing (ascending) order
    and no number appears three or more times; otherwise return False.
    """
    # Check non-decreasing order
    for i in range(len(lst) - 1):
        if lst[i] > lst[i + 1]:
            return False

    # Check for triple or more duplicates
    counts = Counter(lst)
    for c in counts.values():
        if c >= 3:
            return False

    return True
