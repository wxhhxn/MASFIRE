def bf(planet1, planet2):
    """
    Return a tuple of planet names whose orbits lie between planet1 and planet2,
    sorted by proximity to the sun. If either planet name is invalid, return
    an empty tuple.
    """
    planets = (
        "Mercury",
        "Venus",
        "Earth",
        "Mars",
        "Jupiter",
        "Saturn",
        "Uranus",
        "Neptune",
    )
    try:
        i1 = planets.index(planet1)
        i2 = planets.index(planet2)
    except ValueError:
        return tuple()
    start = min(i1, i2) + 1
    end = max(i1, i2)
    return planets[start:end]
