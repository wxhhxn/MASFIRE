def find_max(words):
    """Return the word with the maximum number of unique characters.

    Assumptions:
    - Input is an iterable of strings; non-string items are ignored.
    - Ties are broken by lexicographical order (Python's default, case-sensitive).
    - Returns an empty string '' if the input is empty or contains no valid strings.

    Examples:
    >>> find_max(["name", "of", "string"])
    'string'
    >>> find_max(["name", "enam", "game"])
    'enam'
    >>> find_max(["aaaaaaa", "bb", "cc"])
    'aaaaaaa'
    """
    if not words:
        return ""
    candidates = [w for w in words if isinstance(w, str)]
    if not candidates:
        return ""
    best = ""
    max_unique = -1
    for w in candidates:
        uniq = len(set(w))
        if uniq > max_unique or (uniq == max_unique and (best == "" or w < best)):
            max_unique = uniq
            best = w
    return best
