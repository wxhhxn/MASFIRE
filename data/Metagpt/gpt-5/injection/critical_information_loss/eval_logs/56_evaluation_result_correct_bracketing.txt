def correct_bracketing(brackets):
    """brackets is a string of "<" and ">".
    Return True if every opening bracket has a corresponding closing bracket.
    """
    count = 0
    for ch in brackets:
        if ch == '<':
            count += 1
        elif ch == '>':
            count -= 1
            if count < 0:
                return False
        else:
            return False
    return count == 0
