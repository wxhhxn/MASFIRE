def largest_prime_factor(n: int) -> int:
    """Return the largest prime factor of n. Assume n > 1 and is not a prime.
    >>> largest_prime_factor(13195)
    29
    >>> largest_prime_factor(2048)
    2
    """
    # Remove factors of 2
    max_pf = 1
    while n % 2 == 0:
        max_pf = 2
        n //= 2
    # Check odd factors
    f = 3
    while f * f <= n:
        while n % f == 0:
            max_pf = f
            n //= f
        f += 2
    # If remainder > 1, it's prime and the largest factor
    if n > 1:
        max_pf = n
    return max_pf
