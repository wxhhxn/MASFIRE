def add(lst: list[int]) -> int:
    """
    Assumptions:
    - 0-based indexing: "odd indices" are positions 1, 3, 5, ...
    - Include negative even numbers in the sum.
    - Input list is non-empty and contains integers.

    Returns the sum of even elements located at odd indices.

    Example:
        add([4, 2, 6, 7]) -> 2
    """
    return sum(x for i, x in enumerate(lst) if i % 2 == 1 and x % 2 == 0)
