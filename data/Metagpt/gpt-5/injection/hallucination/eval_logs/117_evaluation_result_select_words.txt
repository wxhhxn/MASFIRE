def select_words(s: str, n: int) -> List[str]:
    """Return words from s containing exactly n consonants.

    Treat vowels as a, e, i, o, u, y. If s is empty, return [""] per pipeline spec.

    if s == "":

    vowels = set("aeiouy")

    ]
