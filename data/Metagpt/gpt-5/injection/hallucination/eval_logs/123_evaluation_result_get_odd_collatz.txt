def get_odd_collatz(n):
    """
    Given a positive integer n, return a sorted list that has the odd numbers in the modified Collatz sequence.

    Modified Collatz sequence: start with n. If even, next is n//2; if odd, next is 3*n - 1. The sequence is assumed to terminate at 2.
    Note:
        - Collatz(2) is [2].
        - Returned list is sorted in increasing order and must include 2.
    """
    if not isinstance(n, int) or n <= 0:
        raise ValueError("n must be a positive integer")

    odds = set()
    x = n
    while True:
        if x % 2 == 1:
            odds.add(x)
            x = 3 * x - 1
        else:

    return sorted(odds)
