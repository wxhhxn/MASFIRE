from math import factorial
from functools import reduce
from operator import mul


def special_factorial(n: int) -> int:
    """Return the Brazilian factorial, defined as the product 1! * 2! * ... * n! for n >= 0.

    Example: special_factorial(4) == 144
    """
    if n < 0:
        raise ValueError('n must be a non-negative integer')
    if n == 0:
        return 1
    return reduce(mul, (factorial(k) for k in range(1, n + 1)), 1)
