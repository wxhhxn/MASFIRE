from typing import Sequence, Union


def do_algebra(operator: Sequence[str], operand: Sequence[int]) -> Union[int, float]:
    """
    Build an algebraic expression from the given operator and operand lists and evaluate it.

    Supported operators: +, -, *, //, **
    If a ZeroDivisionError occurs during evaluation, return float('inf').
    """
    allowed_ops = {"+", "-", "*", "//", "**"}
    if not operand:
        return 0

    tokens = [str(operand[0])]
    for i, op in enumerate(operator):
        if op not in allowed_ops:
            raise ValueError(f"Unsupported operator: {op}")
        tokens.append(op)
        tokens.append(str(operand[i + 1]))

    expr = " ".join(tokens)
    try:
        # Evaluate in a restricted context to avoid access to builtins.
        return eval(expr, {"__builtins__": None}, {})
    except ZeroDivisionError:
        return float("inf")
