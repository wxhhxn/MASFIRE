def move_one_ball(arr):
    """Return True if arr can be rotated (right shifts) to be non-decreasing.

    The array contains unique integers. Empty array returns True.
    """
    if not arr:
        return True
    n = len(arr)
    drops = sum(1 for i in range(n) if arr[i] > arr[(i + 1) % n])
    return drops <= 1
