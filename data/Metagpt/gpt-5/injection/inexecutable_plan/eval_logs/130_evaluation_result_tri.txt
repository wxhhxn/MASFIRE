from typing import List

def tri(n: int) -> List[int]:
    """
    Return the list [tri(0), ..., tri(n)] of the Tribonacci sequence defined by:
    tri(0) = 1
    tri(1) = 3
    for even n >= 2: tri(n) = 1 + n // 2
    for odd n >= 3: tri(n) = tri(n - 1) + tri(n - 2) + tri(n + 1), with tri(n + 1) = 1 + (n + 1) // 2
    Computed iteratively; outputs are ints; handles n = 0.
    """
    seq: List[int] = []
    for i in range(n + 1):
        if i == 0:
            seq.append(1)
        elif i == 1:
            seq.append(3)
        elif i % 2 == 0:
            seq.append(1 + i // 2)
        else:
            next_even = 1 + (i + 1) // 2
            seq.append(seq[i - 1] + seq[i - 2] + next_even)
    return seq
