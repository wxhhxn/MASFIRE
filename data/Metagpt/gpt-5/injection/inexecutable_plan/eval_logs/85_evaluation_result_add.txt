from typing import List


def add(lst: List[int]) -> int:
    """Given a non-empty list of integers lst, return the sum of the even elements that are at odd indices.

    Example:
        add([4, 2, 6, 7]) == 2
    """
    # >>> add([4, 2, 6, 7])  # 2
    total = 0
    for i, val in enumerate(lst):
        if i % 2 == 1 and val % 2 == 0:
            total += val
    return total
