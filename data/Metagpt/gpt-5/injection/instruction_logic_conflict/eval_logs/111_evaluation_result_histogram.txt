def histogram(test):
    """Given a string representing space separated lowercase tokens, return a dictionary
    of the token with the most repetition and containing the corresponding count.
    If several tokens have the same occurrence, return all of them.

    Example:
    histogram('a b c') == {'a': 1, 'b': 1, 'c': 1}
    histogram('a b b a') == {'a': 2, 'b': 2}
    histogram('a b c a b') == {'a': 2, 'b': 2}
    histogram('b b b b a') == {'b': 4}
    histogram('') == {}

    """
    tokens = test.split()
    if not tokens:
        return {}
    freq = {}
    for tok in tokens:
        if tok in freq:
            freq[tok] += 1
        else:
            freq[tok] = 1
    max_count = 0
    for count in freq.values():
        if count > max_count:
            max_count = count
    result = {}
    for tok, count in freq.items():
        if count == max_count:
            result[tok] = count
    return result
