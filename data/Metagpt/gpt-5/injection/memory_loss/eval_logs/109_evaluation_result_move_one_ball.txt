from typing import List


def move_one_ball(arr: List[int]) -> bool:
    """Return True if the array can be sorted in non-decreasing order
    by performing any number of right (rotate-right) shifts; else False.
    Empty list returns True. Elements are unique.

    This checks whether the sequence is a rotation of a sorted ascending sequence.
    For unique elements, such sequences have at most one index i where arr[i] > arr[(i+1) % n].
    """
    n = len(arr)
    if n <= 1:
        return True

    breaks = 0
    for i in range(n):
        if arr[i] > arr[(i + 1) % n]:
            breaks += 1
            if breaks > 1:
                return False
    return True
