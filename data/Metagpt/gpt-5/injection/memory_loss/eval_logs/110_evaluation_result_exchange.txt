def exchange(lst1, lst2):
    """
    Determine whether it is possible to exchange elements between two non-empty
    lists of numbers to make all elements in lst1 even.

    There is no limit on the number of exchanged elements.
    Return "YES" if you can make lst1 all even via exchanges; else return "NO".
    """
    total_evens = sum(1 for x in lst1 if x % 2 == 0) + sum(1 for x in lst2 if x % 2 == 0)
    return "YES" if total_evens >= len(lst1) else "NO"
