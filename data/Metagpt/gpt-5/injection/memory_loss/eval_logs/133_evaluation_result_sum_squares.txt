import math


def sum_squares(lst):
    """Return the sum of squares of the ceiling of each number in lst.

    Each element is rounded up using math.ceil before squaring.
    Examples:
    - [1, 2, 3] -> 14
    - [1, 4, 9] -> 98
    - [1, 3, 5, 7] -> 84
    - [1.4, 4.2, 0] -> 29
    - [-2.4, 1, 1] -> 6
    """
    return sum(math.ceil(x) ** 2 for x in lst)
