import math


def poly(xs: list, x: float):
    return sum(coeff * math.pow(x, i) for i, coeff in enumerate(xs))


def find_zero(xs: list):
    if not isinstance(xs, list):
        raise ValueError("xs must be a list")

    # Trim trailing zeros to determine effective degree and leading coefficient
    j = len(xs) - 1
    while j >= 0 and xs[j] == 0:
        j -= 1
    if j < 0:
        raise ValueError("polynomial cannot be identically zero")
    eff = xs[:j + 1]

    m = len(eff)
    if m % 2 != 0:
        raise ValueError("xs must have an even number of coefficients after trimming")

    # Linear case: ax + b = 0 -> x = -b/a, with coeffs [b, a]
    if m == 2:
        return -eff[0] / eff[1]

    def f(val: float) -> float:
        return poly(eff, val)

    # Quick check at 0
    f0 = f(0.0)
    if abs(f0) <= 1e-15:
        return 0.0

    # Symmetric bracketing for odd-degree polynomial
    a = -1.0
    b = 1.0
    fa = f(a)
    fb = f(b)
    k = 0
    max_expand = 64
    while fa * fb > 0 and k < max_expand:
        k += 1
        a = -(2.0 ** k)
        b = 2.0 ** k
        fa = f(a)
        fb = f(b)

    if abs(fa) <= 1e-15:
        return a
    if abs(fb) <= 1e-15:
        return b

    if fa * fb > 0:
        raise ValueError("Failed to bracket a root")

    # Bisection
    left, right = a, b
    fl, fr = fa, fb
    tol = 1e-12
    max_iter = 200
    mid = 0.0
    for _ in range(max_iter):
        mid = (left + right) / 2.0
        fm = f(mid)
        if abs(fm) <= tol:
            return mid
        if fl * fm <= 0:
            right = mid
            fr = fm
        else:
            left = mid
            fl = fm
    return mid
