def correct_bracketing(brackets: str) -> bool:
    """
    brackets is a string of '<' and '>'.
    return True if every opening bracket has a corresponding closing bracket.
    """
    count = 0
    for ch in brackets:
        if ch == '<':
            count += 1
        elif ch == '>':
            if count == 0:
                return False
            count -= 1
    return count == 0
