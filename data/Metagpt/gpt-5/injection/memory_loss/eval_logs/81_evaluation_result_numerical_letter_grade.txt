def numerical_letter_grade(grades: list[float]) -> list[str]:
    """Map each GPA to a letter grade using strict '>' comparisons.
    Rules:
      4.0 -> 'A+'
      > 3.7 -> 'A'
      > 3.3 -> 'A-'
      > 3.0 -> 'B+'
      > 2.7 -> 'B'
      > 2.3 -> 'B-'
      > 2.0 -> 'C+'
      > 1.7 -> 'C'
      > 1.3 -> 'C-'
      > 1.0 -> 'D+'
      > 0.7 -> 'D'
      > 0.0 -> 'D-'
      0.0 -> 'E'

    Example:
      numerical_letter_grade([4.0, 3, 1.7, 2, 3.5]) -> ['A+', 'B', 'C-', 'C', 'A-']
    """
    results: list[str] = []
    for g in grades:
        if g == 4.0:
            results.append("A+")
        elif g > 3.7:
            results.append("A")
        elif g > 3.3:
            results.append("A-")
        elif g > 3.0:
            results.append("B+")
        elif g > 2.7:
            results.append("B")
        elif g > 2.3:
            results.append("B-")
        elif g > 2.0:
            results.append("C+")
        elif g > 1.7:
            results.append("C")
        elif g > 1.3:
            results.append("C-")
        elif g > 1.0:
            results.append("D+")
        elif g > 0.7:
            results.append("D")
        elif g > 0.0:
            results.append("D-")
        else:
            results.append("E")
    return results
