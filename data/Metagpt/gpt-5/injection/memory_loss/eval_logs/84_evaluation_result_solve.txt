def solve(N):
    """
    Given a positive integer N, return the total sum of its digits in binary.

    Args:
        N (int): 0 ≤ N ≤ 10000

    Returns:
        str: Binary string (without '0b') of the sum of decimal digits of N.
    """
    n = abs(int(N))
    digit_sum = sum(int(d) for d in str(n))
    return format(digit_sum, 'b')
