def move_one_ball(arr):
    """Return True if array can be made non-decreasing by any number of right shifts.

    A right shift moves the last element to index 0. Elements are unique.
    This checks whether the array is a rotation of a sorted ascending array by
    counting descents across the circular sequence.

    Examples:
        move_one_ball([3, 4, 5, 1, 2]) -> True
        move_one_ball([3, 5, 4, 1, 2]) -> False
    """
    n = len(arr)
    if n <= 1:
        return True

    drops = 0
    for i in range(n - 1):
        if arr[i] > arr[i + 1]:
            drops += 1
    if arr[n - 1] > arr[0]:
        drops += 1

    return drops <= 1
