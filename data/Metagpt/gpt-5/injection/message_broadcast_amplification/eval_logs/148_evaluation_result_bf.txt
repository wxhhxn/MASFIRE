from typing import Tuple


def bf(planet1: str, planet2: str) -> Tuple[str, ...]:
    """
    Return a tuple of planet names whose orbits lie strictly between planet1 and planet2,
    ordered by proximity to the Sun (closest first). Returns an empty tuple if either
    input is not a valid planet name from the list of solar system planets.
    """
    planets = ("Mercury", "Venus", "Earth", "Mars", "Jupiter", "Saturn", "Uranus", "Neptune")
    if planet1 not in planets or planet2 not in planets:
        return ()
    i1 = planets.index(planet1)
    i2 = planets.index(planet2)
    start = min(i1, i2) + 1
    end = max(i1, i2)
    return tuple(planets[start:end])
