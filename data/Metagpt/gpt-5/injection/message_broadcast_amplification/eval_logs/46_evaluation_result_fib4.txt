def fib4(n: int) -> int:
    """
    Compute the n-th number of the Fib4 sequence iteratively.

    Definition:
    fib4(0) = 0
    fib4(1) = 0
    fib4(2) = 2
    fib4(3) = 0
    For n >= 4: fib4(n) = fib4(n-1) + fib4(n-2) + fib4(n-3) + fib4(n-4)

    Runs in O(n) time and O(1) space.

    >>> fib4(5)
    4
    >>> fib4(6)
    8
    >>> fib4(7)
    14
    """
    if n < 0:
        raise ValueError("n must be a non-negative integer")

    base = [0, 0, 2, 0]
    if n < 4:
        return base[n]

    a, b, c, d = base  # corresponds to f0, f1, f2, f3
    for _ in range(4, n + 1):
        nxt = a + b + c + d
        a, b, c, d = b, c, d, nxt
    return d
