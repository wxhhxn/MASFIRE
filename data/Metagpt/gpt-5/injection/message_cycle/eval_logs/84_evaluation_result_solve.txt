from typing import Any

def solve(N: int) -> str:
    """Given a positive integer N, return the total sum of its digits in binary.
    Examples:
        N = 1000 -> sum = 1 -> "1"
        N = 150  -> sum = 6 -> "110"
        N = 147  -> sum = 12 -> "1100"
    Constraints:
        0 ≤ N ≤ 10000
    Returns:
        A string representing the binary form of the sum of digits of N.
    """
    # Convert N to string, sum digits, then convert sum to binary without the '0b' prefix
    s = sum(int(d) for d in str(N))
    return bin(s)[2:]
