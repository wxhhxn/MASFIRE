def encode(message: str) -> str:
    """
    Encode the message by first swapping the case of all letters, then
    replacing vowels (A, E, I, O, U and a, e, i, o, u) with the letter
    two positions ahead in the English alphabet, preserving case.

    Non-letter characters (e.g., spaces) are left unchanged.

    Examples:
    >>> encode('test')
    'TGST'
    >>> encode('This is a message')
    'tHKS KS C MGSSCGG'
    """
    swapped = message.swapcase()
    mapping = {
        ord('A'): ord('C'),
        ord('E'): ord('G'),
        ord('I'): ord('K'),
        ord('O'): ord('Q'),
        ord('U'): ord('W'),
        ord('a'): ord('c'),
        ord('e'): ord('g'),
        ord('i'): ord('k'),
        ord('o'): ord('q'),
        ord('u'): ord('w'),
    }
    return swapped.translate(mapping)
