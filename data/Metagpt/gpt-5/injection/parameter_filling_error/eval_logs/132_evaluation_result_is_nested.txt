def is_nested(string: str) -> bool:
    """
    Return True iff there exists a valid bracket subsequence with at least one nested pair.
    The subsequence is of the form "[ [ ] ]" (indices i < j < k < l).
    The input string contains only '[' and ']'.
    """
    second_open_idx = -1
    opens = 0
    for i, ch in enumerate(string):
        if ch == '[':
            opens += 1
            if opens == 2:
                second_open_idx = i
                break
    if second_open_idx == -1:
        return False
    closes_after = 0
    for ch in string[second_open_idx + 1:]:
        if ch == ']':
            closes_after += 1
            if closes_after >= 2:
                return True
    return False
