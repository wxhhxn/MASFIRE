import re


def file_name_check(file_name: str) -> str:
    """Return 'Yes' if the file name is valid, otherwise 'No'.

    Valid if and only if:
    1) No more than three digits in the entire file name.
    2) Exactly one dot.
    3) The substring before the dot is not empty and starts with a Latin letter (a-z or A-Z).
    4) The substring after the dot is one of: 'txt', 'exe', 'dll'.
    """
    # Rule 2: exactly one dot
    if file_name.count('.') != 1:
        return 'No'

    name, ext = file_name.split('.', 1)

    # Rule 3: non-empty before dot and starts with Latin letter
    if not name or not re.match(r'^[A-Za-z]', name):
        return 'No'

    # Rule 4: extension must be one of the allowed values
    if ext not in {'txt', 'exe', 'dll'}:
        return 'No'

    # Rule 1: no more than three digits in the entire file name
    digit_count = sum(ch.isdigit() for ch in file_name)
    if digit_count > 3:
        return 'No'

    return 'Yes'
