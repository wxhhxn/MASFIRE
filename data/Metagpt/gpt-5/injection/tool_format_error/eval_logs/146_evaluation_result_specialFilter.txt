def specialFilter(nums: list[int]) -> int:
    """Count numbers > 10 whose first and last digits of |n| are odd (1,3,5,7,9).

    Args:
        nums: List of integers.

    Returns:
        The count of elements n where n > 10 and both the first and last digits of
        the absolute value of n are odd.
    """
    count = 0
    for n in nums:
        if n > 10:
            m = abs(n)
            last = m % 10
            while m >= 10:
                m //= 10
            first = m
            if (first % 2 == 1) and (last % 2 == 1):
                count += 1
    return count
