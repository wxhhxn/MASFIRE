def prime_fib(n: int) -> int:

            return False
        if x % 2 == 0:
            return x == 2
        if x % 3 == 0:
            return x == 3
        i = 5
        w = 2
        while i * i <= x:
            if x % i == 0:
                return False
            i += w
            w = 6 - w
        return True
