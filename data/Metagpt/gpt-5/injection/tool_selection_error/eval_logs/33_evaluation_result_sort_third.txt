def sort_third(l: list) -> list:
    """Return a new list l' where items at indices divisible by 3 (0, 3, 6, ...) are sorted
    while other positions remain unchanged.

    >>> sort_third([1, 2, 3])
    [1, 2, 3]
    >>> sort_third([5, 6, 3, 4, 8, 9, 2])
    [2, 6, 3, 4, 8, 9, 5]
    """
    res = list(l)
    indices = [i for i in range(len(res)) if i % 3 == 0]
    sorted_vals = sorted(res[i] for i in indices)
    for idx, val in zip(indices, sorted_vals):
        res[idx] = val
    return res
